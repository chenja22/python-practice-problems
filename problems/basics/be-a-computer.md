    def mystery(x):
        if x == 1:
            return [1]
        else:
            output = [1]
            prev = mystery(x - 1)
            for i in range(1, len(prev)):
                output.append(prev[i - 1] + prev[i])
            output.append(1)
        return output

    for i in range(1,6):
